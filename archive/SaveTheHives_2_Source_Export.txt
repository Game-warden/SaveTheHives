// === MATH UTILS ===
function degreesToRadians(degrees) { return degrees * Math.PI / 180; }
function radiansToDegrees(radians) { return radians * 180 / Math.PI; }

function computeIntersection(lat1, lon1, brng1, lat2, lon2, brng2) {
    const phi1 = degreesToRadians(lat1), lambda1 = degreesToRadians(lon1);
    const phi2 = degreesToRadians(lat2), lambda2 = degreesToRadians(lon2);
    const theta13 = degreesToRadians(brng1), theta23 = degreesToRadians(brng2);

    const deltaPhi = phi2 - phi1, deltaLambda = lambda2 - lambda1;
    const delta12 = 2 * Math.asin(Math.sqrt(
        Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) +
        Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2)
    ));

    if (delta12 === 0) return null;

    const thetaA = Math.acos((Math.sin(phi2) - Math.sin(phi1) * Math.cos(delta12)) / (Math.sin(delta12) * Math.cos(phi1)));
    const thetaB = Math.acos((Math.sin(phi1) - Math.sin(phi2) * Math.cos(delta12)) / (Math.sin(delta12) * Math.cos(phi2)));

    const theta12 = Math.sin(lambda2 - lambda1) > 0 ? thetaA : 2 * Math.PI - thetaA;
    const theta21 = Math.sin(lambda2 - lambda1) > 0 ? 2 * Math.PI - thetaB : thetaB;

    const alpha1 = theta13 - theta12;
    const alpha2 = theta21 - theta23;

    if (Math.sin(alpha1) === 0 && Math.sin(alpha2) === 0) return null;
    if (Math.sin(alpha1) * Math.sin(alpha2) < 0) return null;

    const alpha3 = Math.acos(-Math.cos(alpha1) * Math.cos(alpha2) + Math.sin(alpha1) * Math.sin(alpha2) * Math.cos(delta12));
    const delta13 = Math.atan2(Math.sin(delta12) * Math.sin(alpha1) * Math.sin(alpha2), Math.cos(alpha2) + Math.cos(alpha1) * Math.cos(alpha3));

    const phi3 = Math.asin(Math.sin(phi1) * Math.cos(delta13) + Math.cos(phi1) * Math.sin(delta13) * Math.cos(theta13));
    const deltaLambda13 = Math.atan2(Math.sin(theta13) * Math.sin(delta13) * Math.cos(phi1), Math.cos(delta13) - Math.sin(phi1) * Math.sin(phi3));

    let lambda3 = lambda1 + deltaLambda13;
    lambda3 = (lambda3 + 3 * Math.PI) % (2 * Math.PI) - Math.PI;

    return { lat: radiansToDegrees(phi3), lng: radiansToDegrees(lambda3) };
}

function calculateDistanceYards(lat1, lon1, lat2, lon2) {
    const R = 6371e3;
    const phi1 = lat1 * Math.PI / 180;
    const phi2 = lat2 * Math.PI / 180;
    const deltaPhi = (lat2 - lat1) * Math.PI / 180;
    const deltaLambda = (lon2 - lon1) * Math.PI / 180;
    const a = Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) + Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return (R * c) * 1.09361; // meters to yards
}

function calculateBearing(lat1, lon1, lat2, lon2) {
    const phi1 = degreesToRadians(lat1), phi2 = degreesToRadians(lat2);
    const deltaLambda = degreesToRadians(lon2 - lon1);
    const y = Math.sin(deltaLambda) * Math.cos(phi2);
    const x = Math.cos(phi1) * Math.sin(phi2) - Math.sin(phi1) * Math.cos(phi2) * Math.cos(deltaLambda);
    return (radiansToDegrees(Math.atan2(y, x)) + 360) % 360;
}

export default function BeeLiningTool({ onCalculate, onClose, onClear, targetIntersection, onSightingsUpdate }) {
    // 0: Init, 1: Sighting A, 2: Move instruct, 3: Sighting B, 4: Tracking Focus
    const [step, setStep] = useState(0);

    // In-memory strictly, no persistence
    const [sightingA, setSightingA] = useState(null);
    const [sightingB, setSightingB] = useState(null);

    const [currentPos, setCurrentPos] = useState(null);
    const [heading, setHeading] = useState(0);
    const [error, setError] = useState('');

    const walkDistance = sightingA && currentPos ? calculateDistanceYards(sightingA.lat, sightingA.lng, currentPos.lat, currentPos.lng) : 0;

    const watchId = useRef(null);

    // If an external target is passed and we aren't tracking, jump to tracking
    useEffect(() => {
        if (targetIntersection && step < 4) {
            setStep(4);
            startSensors();
        }
    }, [targetIntersection]);

    // Send Sightings to App for Map Render
    useEffect(() => {
        if (onSightingsUpdate) {
            onSightingsUpdate({ a: sightingA, b: sightingB, current: currentPos });
        }
    }, [sightingA, sightingB, currentPos, onSightingsUpdate]);

    const handleOrientation = useCallback((e) => {
        let h = null;
        if (e.webkitCompassHeading) h = e.webkitCompassHeading;
        else if (e.alpha !== null) h = 360 - e.alpha;
        if (h !== null) setHeading(h);
    }, []);

    const handlePosition = useCallback((pos) => {
        setCurrentPos({ lat: pos.coords.latitude, lng: pos.coords.longitude });
    }, []);

    const startSensors = () => {
        if (typeof DeviceOrientationEvent !== 'undefined' && typeof DeviceOrientationEvent.requestPermission === 'function') {
            DeviceOrientationEvent.requestPermission()
                .then(response => {
                    if (response === 'granted') {
                        window.addEventListener('deviceorientation', handleOrientation);
                    } else {
                        setError('Compass denied. Cannot use wizard.');
                    }
                })
                .catch(err => {
                    window.addEventListener('deviceorientation', handleOrientation);
                });
        } else {
            window.addEventListener('deviceorientation', handleOrientation);
        }

        if (navigator.geolocation) {
            watchId.current = navigator.geolocation.watchPosition(handlePosition, (e) => { }, { enableHighAccuracy: true });
        }
        if (step === 0 && !targetIntersection) setStep(1);
    };

    const stopSensors = () => {
        window.removeEventListener('deviceorientation', handleOrientation);
        if (watchId.current) {
            navigator.geolocation.clearWatch(watchId.current);
            watchId.current = null;
        }
    };

    useEffect(() => {
        return () => stopSensors(); // Cleanup on unmount
    }, []);

    const handleClear = () => {
        setSightingA(null);
        setSightingB(null);
        setStep(0);
        setError('');
        stopSensors();
        if (onClear) onClear();
        if (onSightingsUpdate) onSightingsUpdate({ a: null, b: null, current: null });
    };

    const captureA = () => {
        if (!currentPos) return;
        setSightingA({ lat: currentPos.lat, lng: currentPos.lng, bearing: heading });
        console.log(`[BeeHunt Wizard] Sighting A Captured:`, { lat: currentPos.lat, lng: currentPos.lng, bearing: heading });
        setStep(2);
    };

    const captureB = () => {
        if (!currentPos || !sightingA) return;
        const b = { lat: currentPos.lat, lng: currentPos.lng, bearing: heading };
        setSightingB(b);
        console.log(`[BeeHunt Wizard] Sighting B Captured:`, b);

        const intersect = computeIntersection(sightingA.lat, sightingA.lng, sightingA.bearing, b.lat, b.lng, b.bearing);
        if (!intersect) {
            console.error(`[BeeHunt Wizard] Intersection failed. Lines don't cross.`);
            setError("Lines don't cross! Try tracking again.");
            setStep(0);
            return;
        }
        console.log(`[BeeHunt Wizard] Target Intersection Calculated:`, intersect);
        onCalculate(intersect);
        setStep(4);
    };

    return (
        <div className="mt-4 bg-[#7d6d52]/5 p-4 rounded-xl border border-[#7d6d52]/20 shadow-inner">
            <h4 className="font-bold text-sm text-[#7d6d52] mb-3 flex items-center justify-between">
                Automated Bee Hunt
                <span className="bg-red-500/10 text-red-600 px-2 py-0.5 rounded text-[8px] uppercase tracking-wider">Privacy First</span>
            </h4>

            {error && <p className="text-red-500 text-xs font-bold bg-white p-2 rounded shadow-sm mb-3">{error}</p>}

            {step === 0 && (
                <div className="space-y-3">
                    <p className="text-xs text-[#7d6d52]/80 leading-relaxed italic">Use your phone as a compass. We will triangulate the feral hive with zero data scraping.</p>
                    <button onClick={startSensors} className="w-full bg-[#7d6d52] text-[#f4f1ea] py-3 rounded-lg text-sm font-bold shadow-md active:scale-95 transition-transform">
                        Grant Compass & GPS
                    </button>
                </div>
            )}

            {step === 1 && (
                <div className="space-y-4 text-center">
                    <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                        <p className="text-[10px] font-black tracking-widest uppercase text-[#7d6d52]/60 mb-1">Magnetic Heading</p>
                        <p className="text-3xl font-mono text-[#7d6d52] opacity-90">{Math.round(heading)}°</p>
                    </div>
                    <p className="text-xs text-[#7d6d52] font-semibold">Face the bees' flight path exactly.</p>
                    <button onClick={captureA} disabled={!currentPos} className="w-full bg-[#3b82f6] text-white py-3 rounded-lg text-sm font-bold shadow-md active:scale-95 transition-transform disabled:opacity-50">
                        {currentPos ? "Capture Sighting A" : "Acquiring GPS..."}
                    </button>
                </div>
            )}

            {step === 2 && (
                <div className="space-y-4 text-center">
                    <div className="bg-[#3b82f6]/10 p-4 rounded-xl border border-[#3b82f6]/30 shadow-sm">
                        <p className="text-xs font-bold text-[#3b82f6] uppercase tracking-wider mb-2">Distance from A</p>
                        <p className="text-3xl font-black text-[#7d6d52]">
                            {Math.round(walkDistance)}
                            <span className="text-sm font-medium text-[#7d6d52]/50 ml-1">yds</span>
                        </p>
                        <div className="w-full bg-gray-200 rounded-full h-1.5 mb-2 mt-3">
                            <div className="bg-[#3b82f6] h-1.5 rounded-full transition-all duration-300" style={{ width: `${Math.min((walkDistance / 20) * 100, 100)}%` }}></div>
                        </div>
                    </div>

                    <p className="text-xs text-[#7d6d52] font-semibold bg-white p-3 rounded-lg border border-gray-100 shadow-sm">
                        Sighting A locked. Walk 20 yards along the blue dashed line.
                    </p>

                    <button
                        onClick={() => setStep(3)}
                        className={`w-full py-3 rounded-lg text-sm font-bold shadow-md active:scale-95 transition-all ${walkDistance >= 20 ? 'bg-[#16a34a] text-white animate-pulse' : 'bg-[#7d6d52]/50 text-white'}`}
                    >
                        {walkDistance >= 20 ? "I'm in Position" : "Override: I'm in Position"}
                    </button>
                </div>
            )}

            {step === 3 && (
                <div className="space-y-4 text-center">
                    <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                        <p className="text-[10px] font-black tracking-widest uppercase text-[#7d6d52]/60 mb-1">Magnetic Heading</p>
                        <p className="text-3xl font-mono text-[#7d6d52] opacity-90">{Math.round(heading)}°</p>
                    </div>

                    <p className="text-xs text-[#7d6d52] font-semibold bg-white p-3 rounded-lg border border-gray-100 shadow-sm">
                        20 yards reached. Turn to face in approximately the same direction the first path was. Release bees and capture vanishing path B.
                    </p>

                    <button onClick={captureB} disabled={!currentPos} className="w-full bg-[#16a34a] text-white py-3 rounded-lg text-sm font-bold shadow-md active:scale-95 transition-transform disabled:opacity-50">
                        Capture Sighting B
                    </button>
                    <button onClick={() => setStep(1)} className="w-full bg-transparent text-[#7d6d52] text-xs font-bold py-2 underline opacity-60">Redo Sighting A</button>
                </div>
            )}

            {step === 4 && targetIntersection && currentPos && (
                <div className="space-y-4">
                    <div className="bg-white p-4 rounded-xl shadow-md border-2 border-[#8b5cf6]/50 flex flex-col items-center">
                        <p className="text-[10px] font-black tracking-widest uppercase text-[#8b5cf6] mb-2">Distance to Target</p>
                        <p className="text-4xl font-black text-[#374151]">
                            {Math.round(calculateDistanceYards(currentPos.lat, currentPos.lng, targetIntersection.lat, targetIntersection.lng))}
                            <span className="text-lg text-gray-400 ml-1">yds</span>
                        </p>

                        <div className="relative mt-4 w-16 h-16 rounded-full bg-gray-100 border border-gray-200 flex items-center justify-center">
                            <div
                                style={{ transform: `rotate(${calculateBearing(currentPos.lat, currentPos.lng, targetIntersection.lat, targetIntersection.lng) - heading}deg)` }}
                                className="text-2xl text-[#8b5cf6] transition-transform duration-200 ease-out"
                            >
                                ⬆
                            </div>
                        </div>
                        <p className="text-[10px] text-gray-400 uppercase tracking-widest mt-2">Walk This Way</p>
                    </div>
                </div>
            )}

            {/* Clear Button - Wipes Memory */}
            <div className="flex flex-col space-y-2 pt-4 mt-2 border-t border-[#7d6d52]/10">
                {step > 0 && (
                    <button type="button" onClick={handleClear} className="w-full bg-red-500/10 text-red-600 py-2.5 rounded text-sm font-bold active:scale-95 transition-transform border border-red-500/20">

        </div>
    );
}

// Isolated SearchBar to prevent typing lag in the main App (and thereby the Map)
function SearchBar({ onSearch }) {
  const [val, setVal] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    if (val.length === 5) {
      onSearch(val);
      document.activeElement.blur(); // Hides keyboard
    }
  };

  return (
    <form onSubmit={handleSubmit} className="flex items-center bg-white/10 rounded-lg overflow-hidden border border-white/20">
      <input
        type="number"
        inputMode="numeric"
        placeholder="ZIP"
        className="w-20 bg-transparent text-white px-3 py-2 text-sm outline-none placeholder-white/40"
        style={{ fontSize: '16px' }}
        value={val}
        onChange={(e) => setVal(e.target.value)}
      />
      <button type="submit" className="bg-[#7d6d52] px-3 py-2 text-[#f4f1ea] border-l border-white/10 active:bg-white/30">
        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">

        </svg>
      </button>
    </form>
  );
}

function FilterGroup({ title, items, onToggle, isType }) {
  const getColorDot = (key) => {
    if (isType) {
      const colors = { building: 'bg-blue-500', tree: 'bg-green-600', ground: 'bg-orange-500', beehive: 'bg-purple-500' };
      return colors[key];
    } else {
      const colors = { healthy: 'bg-green-500', warning: 'bg-yellow-500', atRisk: 'bg-red-500' };
      return colors[key];
    }
  };

  return (
    <div className="flex flex-col space-y-1.5">
      <p className="text-[8px] font-black uppercase text-[#7d6d52] text-center border-b border-[#7d6d52]/20 pb-1">{title}</p>
      {Object.keys(items).map(k => (
        <button key={k} onClick={() => onToggle(k)}
          className={`flex items-center px-1.5 py-1.5 rounded text-[8px] font-bold border transition-all outline-none active:scale-95 ${items[k]
            ? 'bg-[#7d6d52] text-white border-[#7d6d52] shadow-sm'
            : 'bg-white/80 text-[#7d6d52]/50 border-gray-200'
            }`}
        >
          <span className={`w-2 h-2 rounded-full flex-shrink-0 mr-1.5 ${getColorDot(k)}`}></span>
          <span className="truncate">{k === 'atRisk' ? 'At-Risk' : k.toUpperCase()}</span>
        </button>
      ))}
    </div>
  );
}

const getHiveCategory = (n = '', s = '') => {
    const text = (n + ' ' + s).toLowerCase();

    // 1. BRACKET SUPREMACY [cite: 31-32]
    const bracketMatch = text.match(/\[(.*?)\]/);
    if (bracketMatch) {
        const b = bracketMatch[1];
        if (b.includes('tree') || b.includes('wood')) return 'tree';
        if (b.includes('ground') || b.includes('earth')) return 'ground';
        if (b.includes('beehive') || b.includes('box') || b.includes('nuc') || b.includes('langstroth')) return 'beehive';
        if (b.includes('building') || b.includes('manmade') || b.includes('structure')) return 'building';
    }

    // 2. PRIORITY FLIP: Trees and Ground First
    if (text.includes('tree') || text.includes('wood') || text.includes('oak') || text.includes('pine')) {
        return 'tree';
    }
    if (text.includes('ground') || text.includes('earth') || text.includes('dirt')) {
        return 'ground';
    }

    // 3. HUMAN STRUCTURES (Building)
    if (text.includes('building') || text.includes('manmade') || text.includes('structure') ||
        text.includes('house') || text.includes('roof') || text.includes('wall')) {
        return 'building';
    }

    // 4. SPECIFIC BEEKEEPING EQUIPMENT (Removed generic 'hive')
    if (text.includes('beehive') || text.includes('box') ||
        text.includes('langstroth') || text.includes('nuc') || text.includes('feral')) {
        return 'beehive';
    }

    return 'beehive';
};

const getRingColor = (s) => ({ healthy: '#22c55e', warning: '#eab308', atRisk: '#ef4444' }[s] || '#eab308');
const getPointerColor = (cat) => ({ building: '#3b82f6', tree: '#16a34a', ground: '#f97316', beehive: '#8b5cf6' }[cat] || '#8b5cf6');

export default function MapComponent({ healthFilters, typeFilters, navigation, onUpdateCounts, targetIntersection, activeSightings }) {
    const mapContainer = useRef(null);
    const map = useRef(null);
    const markersRef = useRef([]);
    const [mapReady, setMapReady] = useState(false);

    useEffect(() => {
        if (map.current) return;

        map.current = new maplibregl.Map({
            container: mapContainer.current,
            style: 'https://basemaps.cartocdn.com/gl/positron-gl-style/style.json',
            center: [-98.5795, 39.8283], // [Lng, Lat] standard 
            zoom: 3,
            attributionControl: false
        });

        // Prevent iOS rubber-banding [cite: 19]
        mapContainer.current.addEventListener('touchmove', (e) => e.preventDefault(), { passive: false });

        map.current.on('load', () => {
            map.current.addControl(new maplibregl.NavigationControl(), 'top-right');
            setMapReady(true);
        });

        map.current.on('moveend', updateCounts);
        map.current.on('zoomend', updateCounts);

        return () => {
            if (map.current) {
                map.current.remove();
                map.current = null;
            }
        };
    }, []);

    useEffect(() => {
        if (!mapReady || !map.current) return;

        markersRef.current.forEach(marker => marker.remove());
        markersRef.current = [];

        if (map.current.getLayer('foraging-rings')) map.current.removeLayer('foraging-rings');
        if (map.current.getSource('foraging-data')) map.current.removeSource('foraging-data');
        if (map.current.getLayer('sighting-a-line')) map.current.removeLayer('sighting-a-line');
        if (map.current.getLayer('sighting-guide-line')) map.current.removeLayer('sighting-guide-line');
        if (map.current.getSource('sighting-lines')) map.current.removeSource('sighting-lines');

        const geojsonFeatures = [];
        let renderedCount = 0;
        let viewableCount = 0;
        const bounds = map.current.getBounds();

        allHives.forEach(h => {
            const cat = getHiveCategory(h.note, h.status);
            if (healthFilters[h.status] && typeFilters[cat]) {
                renderedCount++;
                if (bounds.contains([h.lng, h.lat])) viewableCount++;

                const el = document.createElement('div');
                el.className = 'custom-marker';
                el.style.width = '16px';
                el.style.height = '16px';
                el.style.borderRadius = '50%';
                el.style.backgroundColor = getPointerColor(cat);
                el.style.border = '2px solid white';
                el.style.boxShadow = '0 2px 4px rgba(0,0,0,0.3)';

                const htmlPopup = `
                    <div style="font-family:sans-serif;font-size:12px;min-width:150px;color:#374151;">
                        <div style="display:flex;justify-content:space-between;border-bottom:1px solid #eee;padding-bottom:4px;margin-bottom:4px;font-weight:bold;font-size:10px;text-transform:uppercase;color:#7d6d52;opacity:0.8;">
                            <span>${cat}</span>
                            <span style="color:${getRingColor(h.status)}">${h.status}</span>
                        </div>
                        <b>Date:</b> ${h.date || 'Legacy'}<br/>
                        <div style="margin-top:8px;border-top:1px solid #f3f4f6;padding-top:4px;font-style:italic;font-size:11px;">"${h.note}"</div>
                    </div>
                `;

                const marker = new maplibregl.Marker({ element: el })
                    .setLngLat([h.lng, h.lat])
                    .setPopup(new maplibregl.Popup({ offset: 15, closeButton: false }).setHTML(htmlPopup))
                    .addTo(map.current);
                markersRef.current.push(marker);

                geojsonFeatures.push({
                    type: 'Feature',
                    geometry: { type: 'Point', coordinates: [h.lng, h.lat] },
                    properties: { color: getRingColor(h.status) }
                });
            }
        });

        // Add the Pulsating Target Marker if Triangulated
        if (targetIntersection) {
            const el = document.createElement('div');
            el.className = 'pulsating-target';
            el.style.width = '24px';
            el.style.height = '24px';
            el.style.borderRadius = '50%';
            el.style.backgroundColor = 'rgba(139, 92, 246, 0.4)';
            el.style.border = '3px solid #8b5cf6';
            el.style.animation = 'pulseTarget 1.5s infinite';
            el.style.zIndex = '9999';

            const htmlPopup = `<div style="font-family:sans-serif;font-size:12px;color:#374151;min-width:140px;text-align:center;"><b>🎯 Extrapolated Target</b><div style="font-family:monospace;margin-top:4px;color:#8b5cf6;">Lng: ${targetIntersection.lng.toFixed(5)}<br/>Lat: ${targetIntersection.lat.toFixed(5)}</div></div>`;

            const marker = new maplibregl.Marker({ element: el })
                .setLngLat([targetIntersection.lng, targetIntersection.lat])
                .setPopup(new maplibregl.Popup({ offset: 15, closeButton: false }).setHTML(htmlPopup))
                .addTo(map.current);
            markersRef.current.push(marker);
        }

        map.current.addSource('foraging-data', {
            type: 'geojson',
            data: { type: 'FeatureCollection', features: geojsonFeatures }
        });

        map.current.addLayer({
            id: 'foraging-rings',
            type: 'circle',
            source: 'foraging-data',
            paint: {
                'circle-radius': [
                    'interpolate', ['exponential', 2], ['zoom'],
                    2, 2,
                    10, 40,
                    15, 1200
                ],
                'circle-color': ['get', 'color'],
                'circle-opacity': 0.2,
                'circle-stroke-color': ['get', 'color'],
                'circle-stroke-width': 2
            }
        });

        // --- Render Sighting Lines (Automated Bee Hunt) ---
        const lineFeatures = [];

        // Helper to project a point given lat/lng, bearing, and distance in meters
        const projectPoint = (lat, lng, brng, dist) => {
            const R = 6371e3;
            const brngRad = brng * Math.PI / 180;
            const latRad = lat * Math.PI / 180;
            const lngRad = lng * Math.PI / 180;
            const lat2 = Math.asin(Math.sin(latRad) * Math.cos(dist / R) + Math.cos(latRad) * Math.sin(dist / R) * Math.cos(brngRad));
            const lng2 = lngRad + Math.atan2(Math.sin(brngRad) * Math.sin(dist / R) * Math.cos(latRad), Math.cos(dist / R) - Math.sin(latRad) * Math.sin(lat2));
            return [(lng2 * 180 / Math.PI), (lat2 * 180 / Math.PI)]; // [lng, lat]
        };

        if (activeSightings && activeSightings.a) {
            // Sighting A Solid Green Ray (5000 meters out)
            const endPointA = projectPoint(activeSightings.a.lat, activeSightings.a.lng, activeSightings.a.bearing, 5000);
            lineFeatures.push({
                type: 'Feature',
                properties: { type: 'sighting-a' },
                geometry: {
                    type: 'LineString',
                    coordinates: [[activeSightings.a.lng, activeSightings.a.lat], endPointA]
                }
            });

            // Perpendicular Blue Walk Guide Line (20 yards = ~18 meters)
            const perpBearing = (activeSightings.a.bearing + 90) % 360;
            const endPointGuide = projectPoint(activeSightings.a.lat, activeSightings.a.lng, perpBearing, 18);
            lineFeatures.push({
                type: 'Feature',
                properties: { type: 'sighting-guide' },
                geometry: {
                    type: 'LineString',
                    coordinates: [[activeSightings.a.lng, activeSightings.a.lat], endPointGuide]
                }
            });
        }

        if (lineFeatures.length > 0) {
            map.current.addSource('sighting-lines', {
                type: 'geojson',
                data: { type: 'FeatureCollection', features: lineFeatures }
            });

            map.current.addLayer({
                id: 'sighting-a-line',
                type: 'line',
                source: 'sighting-lines',
                filter: ['==', 'type', 'sighting-a'],
                paint: {
                    'line-color': '#16a34a',
                    'line-width': 3,
                    'line-opacity': 0.8
                }
            });

            map.current.addLayer({
                id: 'sighting-guide-line',
                type: 'line',
                source: 'sighting-lines',
                filter: ['==', 'type', 'sighting-guide'],
                paint: {
                    'line-color': '#3b82f6',
                    'line-width': 3,
                    'line-dasharray': [2, 2],
                    'line-opacity': 0.8
                }
            });
        }

        onUpdateCounts({ total: renderedCount, viewable: viewableCount });
    }, [healthFilters, typeFilters, targetIntersection, activeSightings, mapReady]);

    // Adjusted Zoom Levels for Navigation 
    useEffect(() => {
        if (!navigation || !mapReady || !map.current) return;

        if (navigation.type === 'zip') {
            fetch(`https://nominatim.openstreetmap.org/search?format=json&postalcode=${navigation.value}&country=USA`)
                .then(res => res.json())
                .then(data => {
                    if (data.length > 0) {
                        // Zip Zoom decreased to 10 for better context
                        map.current.flyTo({ center: [parseFloat(data[0].lon), parseFloat(data[0].lat)], zoom: 10 });
                    }
                });
        } else if (navigation.type === 'gps') {
            navigator.geolocation.getCurrentPosition(
                (pos) => {
                    // GPS Zoom decreased to 11 for better context
                    map.current.flyTo({ center: [pos.coords.longitude, pos.coords.latitude], zoom: 11 });
                },
                (error) => console.error("GPS Error:", error),
                { enableHighAccuracy: true, timeout: 5000, maximumAge: 0 }
            );
        } else if (navigation.type === 'target') {
            map.current.flyTo({ center: [navigation.value.lng, navigation.value.lat], zoom: 14 });
        }
    }, [navigation, mapReady]);

    const updateCounts = () => {
        if (!mapReady || !map.current) return;
        const bounds = map.current.getBounds();
        let total = 0;
        let viewable = 0;
        allHives.forEach(h => {
            const cat = getHiveCategory(h.note, h.status);
            if (healthFilters[h.status] && typeFilters[cat]) {
                total++;
                if (bounds.contains([h.lng, h.lat])) viewable++;
            }
        });
        onUpdateCounts({ total, viewable });
    };

    return (
        <div className="absolute inset-0 w-full h-full">
            <div ref={mapContainer} style={{ width: '100%', height: '100%', position: 'absolute' }} />
        </div>
    );
}
