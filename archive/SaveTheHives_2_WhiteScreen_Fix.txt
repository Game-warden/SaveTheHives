import React, { useEffect, useRef, useState } from 'react';
import maplibregl from 'maplibre-gl';
import 'maplibre-gl/dist/maplibre-gl.css';
import * as HiveDataModule from '../hiveData';

const allHives = HiveDataModule.allHives || [];

const getHiveCategory = (n = '', s = '') => {
    const text = (n + ' ' + s).toLowerCase();

    // 1. BRACKET SUPREMACY
    const bracketMatch = text.match(/\[(.*?)\]/);
    if (bracketMatch) {
        const b = bracketMatch[1];
        if (b.includes('tree') || b.includes('wood')) return 'tree';
        if (b.includes('ground') || b.includes('earth')) return 'ground';
        if (b.includes('beehive') || b.includes('box') || b.includes('nuc') || b.includes('langstroth')) return 'beehive';
        if (b.includes('building') || b.includes('manmade') || b.includes('structure')) return 'building';
    }

    // 2. PRIORITY FLIP: Trees and Ground First
    if (text.includes('tree') || text.includes('wood') || text.includes('oak') || text.includes('pine')) {
        return 'tree';
    }
    if (text.includes('ground') || text.includes('earth') || text.includes('dirt')) {
        return 'ground';
    }

    // 3. HUMAN STRUCTURES (Building)
    if (text.includes('building') || text.includes('manmade') || text.includes('structure') ||
        text.includes('house') || text.includes('roof') || text.includes('wall')) {
        return 'building';
    }

    // 4. SPECIFIC BEEKEEPING EQUIPMENT (Removed generic 'hive')
    if (text.includes('beehive') || text.includes('box') ||
        text.includes('langstroth') || text.includes('nuc') || text.includes('feral')) {
        return 'beehive';
    }

    // FALLBACK
    return 'beehive';
};

const getRingColor = (s) => ({ healthy: '#22c55e', warning: '#eab308', atRisk: '#ef4444' }[s] || '#eab308');
const getPointerColor = (cat) => ({ building: '#3b82f6', tree: '#16a34a', ground: '#f97316', beehive: '#8b5cf6' }[cat] || '#8b5cf6');

export default function MapComponent({ healthFilters, typeFilters, navigation, onUpdateCounts }) {
    const mapContainer = useRef(null);
    const map = useRef(null);
    const markersRef = useRef([]);
    const [mapReady, setMapReady] = useState(false);

    // Initialize map
    useEffect(() => {
        if (map.current) return;

        map.current = new maplibregl.Map({
            container: mapContainer.current,
            style: 'https://basemaps.cartocdn.com/gl/positron-gl-style/style.json', // A clean, fast vector tile basemap
            center: [-98.5795, 39.8283], // [Lng, Lat] for USA center
            zoom: 3,
            attributionControl: false
        });

        // Prevent iOS rubber-banding inside the canvas
        mapContainer.current.addEventListener('touchmove', (e) => e.preventDefault(), { passive: false });

        map.current.on('load', () => {
            map.current.addControl(new maplibregl.NavigationControl(), 'top-right');
            setMapReady(true);
        });

        map.current.on('moveend', updateCounts);
        map.current.on('zoomend', updateCounts);

        return () => {
            if (map.current) {
                map.current.remove();
                map.current = null;
            }
        };
    }, []);

    // Handle render updates based on filters
    useEffect(() => {
        if (!mapReady || !map.current) return;

        // Clear existing markers
        markersRef.current.forEach(marker => marker.remove());
        markersRef.current = [];

        // Remove old circle layer and source if they exist
        if (map.current.getLayer('foraging-rings')) {
            map.current.removeLayer('foraging-rings');
        }
        if (map.current.getSource('foraging-data')) {
            map.current.removeSource('foraging-data');
        }

        let renderedCount = 0;
        let viewableCount = 0;
        const bounds = map.current.getBounds();

        // Prepare raw GeoJSON for circles (using points + circle-radius for extreme performance)
        const geojsonFeatures = [];

        allHives.forEach(h => {
            const cat = getHiveCategory(h.note, h.status);

            if (healthFilters[h.status] && typeFilters[cat]) {
                renderedCount++;

                // MapLibre bounds.contains expects LngLat or [Lng, Lat]
                if (bounds.contains([h.lng, h.lat])) {
                    viewableCount++;
                }

                // Create custom HTML marker
                const el = document.createElement('div');
                el.className = 'custom-marker';
                el.style.width = '16px';
                el.style.height = '16px';
                el.style.borderRadius = '50%';
                el.style.backgroundColor = getPointerColor(cat);
                el.style.border = '2px solid white';
                el.style.boxShadow = '0 2px 4px rgba(0,0,0,0.3)';

                const htmlPopup = `
                    <div style="font-family:sans-serif;font-size:12px;min-width:150px;color:#374151;">
                        <div style="display:flex;justify-content:space-between;border-bottom:1px solid #eee;padding-bottom:4px;margin-bottom:4px;font-weight:bold;font-size:10px;text-transform:uppercase;color:#7d6d52;opacity:0.8;">
                            <span>${cat}</span>
                            <span style="color:${getRingColor(h.status)}">${h.status}</span>
                        </div>
                        <b>Date:</b> ${h.date || 'Legacy'}<br/>
                        <div style="margin-top:8px;border-top:1px solid #f3f4f6;padding-top:4px;font-style:italic;font-size:11px;">"${h.note}"</div>
                    </div>
                `;

                const marker = new maplibregl.Marker({ element: el })
                    .setLngLat([h.lng, h.lat])
                    .setPopup(new maplibregl.Popup({ offset: 15, closeButton: false }).setHTML(htmlPopup))
                    .addTo(map.current);

                markersRef.current.push(marker);

                // Add to source for foraging rings
                geojsonFeatures.push({
                    type: 'Feature',
                    geometry: { type: 'Point', coordinates: [h.lng, h.lat] },
                    properties: { color: getRingColor(h.status) }
                });
            }
        });

        // Add the GeoJSON source and layer for blazing fast WebGL circle rendering
        map.current.addSource('foraging-data', {
            type: 'geojson',
            data: {
                type: 'FeatureCollection',
                features: geojsonFeatures
            }
        });

        map.current.addLayer({
            id: 'foraging-rings',
            type: 'circle',
            source: 'foraging-data',
            paint: {
                'circle-radius': [
                    'interpolate', ['exponential', 2], ['zoom'],
                    0, 0,
                    20, 2048 // Scales the conceptual 4828m visually based on zoom
                ],
                'circle-color': ['get', 'color'],
                'circle-opacity': 0.1,
                'circle-stroke-color': ['get', 'color'],
                'circle-stroke-width': 1
            }
        });

        onUpdateCounts({ total: renderedCount, viewable: viewableCount });

    }, [healthFilters, typeFilters, mapReady]);

    // Handle Navigation Commands
    useEffect(() => {
        if (!navigation || !mapReady || !map.current) return;

        if (navigation.type === 'zip') {
            fetch(`https://nominatim.openstreetmap.org/search?format=json&postalcode=${navigation.value}&country=USA`)
                .then(res => res.json())
                .then(data => {
                    if (data.length > 0) {
                        map.current.flyTo({ center: [parseFloat(data[0].lon), parseFloat(data[0].lat)], zoom: 12 });
                    }
                });
        } else if (navigation.type === 'gps') {
            navigator.geolocation.getCurrentPosition((pos) => {
                map.current.flyTo({ center: [pos.coords.longitude, pos.coords.latitude], zoom: 13 });
            });
        }
    }, [navigation, mapReady]);

    const updateCounts = () => {
        if (!mapReady || !map.current) return;
        const bounds = map.current.getBounds();
        let total = 0;
        let viewable = 0;

        allHives.forEach(h => {
            const cat = getHiveCategory(h.note, h.status);
            if (healthFilters[h.status] && typeFilters[cat]) {
                total++;
                if (bounds.contains([h.lng, h.lat])) {
                    viewable++;
                }
            }
        });
        onUpdateCounts({ total, viewable });
    };

    return (
        <div className="absolute inset-0 w-full h-full">
            <div ref={mapContainer} style={{ width: '100%', height: '100%', position: 'absolute' }} />
        </div>
    );
}

const CACHE_NAME = 'savethehives-cache-v2';

// Core assets to cache immediately on install
const urlsToCache = [
    '/',
    '/index.html',
    '/icon.png',
    'https://unpkg.com/maplibre-gl@latest/dist/maplibre-gl.css'
];

// Install Phase: Cache App Shell
self.addEventListener('install', event => {
    event.waitUntil(
        caches.open(CACHE_NAME)
            .then(cache => {
                return cache.addAll(urlsToCache);
            })
    );
    self.skipWaiting();
});

// Activate Phase: Clean up old caches
self.addEventListener('activate', event => {
    const cacheWhitelist = [CACHE_NAME];
    event.waitUntil(
        caches.keys().then(cacheNames => {
            return Promise.all(
                cacheNames.map(cacheName => {
                    if (cacheWhitelist.indexOf(cacheName) === -1) {
                        return caches.delete(cacheName);
                    }
                })
            );
        })
    );
    self.clients.claim();
});

// Fetch Phase: Cache-First Strategy for Offline Rural Connectivity
self.addEventListener('fetch', event => {
    // Only cache GET requests
    if (event.request.method !== 'GET') return;

    event.respondWith(
        caches.match(event.request)
            .then(cachedResponse => {
                if (cachedResponse) {
                    // Serve from cache immediately for speed/offline access
                    // Optional: Fetch network version in background to keep cache fresh
                    event.waitUntil(
                        fetch(event.request).then(networkResponse => {
                            if (networkResponse && networkResponse.status === 200) {
                                caches.open(CACHE_NAME).then(cache => {
                                    cache.put(event.request, networkResponse.clone());
                                });
                            }
                        }).catch(() => { }) // Ignore background network failures
                    );
                    return cachedResponse;
                }

                // If not in cache, fallback to Network
                return fetch(event.request)
                    .then(networkResponse => {
                        // Validate response before caching
                        if (!networkResponse || networkResponse.status !== 200 || networkResponse.type !== 'basic') {
                            return networkResponse;
                        }

                        // Clone the response because it's a stream
                        const responseToCache = networkResponse.clone();
                        caches.open(CACHE_NAME).then(cache => {
                            cache.put(event.request, responseToCache);
                        });

                        return networkResponse;
                    });
            })
    );
});
